SELECT ?title
WHERE {
  ?book :author ?author .
  ?book :title ?title .
  FILTER(CONTAINS(LCASE(?author), "austen"))
}
ORDER BY ?title

SELECT ?genre (COUNT(?book) AS ?numberOfBooks)
WHERE {
  ?book :genres ?genre .
}
GROUP BY ?genre
ORDER BY ?genre
