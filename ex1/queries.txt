SELECT ?nome ?curso ?universidade ?nacionalidade ?amigo
WHERE {
  :Hannah :nome ?nome .
  :Hannah :estudanteFrequentaCurso ?curso .
  :Hannah :pessoaEstaNaUniversidade ?universidade .
  :Hannah :temNacionalidade ?nacionalidade .
  OPTIONAL { :Hannah :eAmigoDe ?amigo . }
}
